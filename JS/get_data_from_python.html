// JavaScript code function getDataFromPythonScript() { const xhr = new XMLHttpRequest(); xhr.open('GET', '/path/to/python/script.py', true); xhr.onreadystatechange = function() { if (xhr.readyState === 4 && xhr.status === 200) { const data = JSON.parse(xhr.responseText); console.log(data); // Do something with the data } }; xhr.send(); } // HTML file Get Data from Python Script
